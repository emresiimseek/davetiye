function updateCountdown() {

  const diff = new Date('2026-09-19T13:00:00') - new Date();

  if (diff <= 0) {

    ['cd-gun','cd-saat','cd-dk','cd-sn'].forEach(id => document.getElementById(id).textContent = '00');

    return;

  }

  document.getElementById('cd-gun').textContent  = String(Math.floor(diff/86400000)).padStart(2,'0');

  document.getElementById('cd-saat').textContent = String(Math.floor((diff%86400000)/3600000)).padStart(2,'0');

  document.getElementById('cd-dk').textContent   = String(Math.floor((diff%3600000)/60000)).padStart(2,'0');

  document.getElementById('cd-sn').textContent   = String(Math.floor((diff%60000)/1000)).padStart(2,'0');
